[Unusual Entertainment](https://codeforces.com/contest/1899/problem/G)

你知道的，我向来ds只会个板子的喵，但是，这是我见到的第二道拿主席树维护二维区间上的问题喵。

这道题的难度在于怎么**转化到二维区间**上的问题喵。

想要判断一个点在不在一个点的子树里面，我们可以学着树上启发式合并，去记录进入这个子树的时间戳和出去的时间戳喵，这样的话，我们只需要判断有没有 $[in[x] \leq in[i] \leq out[x]$ 喵

那么对于上面的这两个条件，我们可以很自然地得出对于每个 $l,r,x$ 就是查询 $[in[x],out[x]] \times [l,r]$ 这个矩形里面有没有点喵

其中的query_reg是在主席树里面定义的喵

```cpp
int query_reg(int top,int bottom,int l,int r){return query(bottom,l,r).val - query(top,l,r).val;}
```

```cpp
void solve()
{	
	int n,q;
	cin>>n>>q;
	vector<vector<int>> g(n + 1);
	for (int i = 1 ; i < n ; i++)
	{
		int u,v;
		cin>>u>>v;
		g[u].emplace_back(v);
		g[v].emplace_back(u);
	}
	vector<int> pmt(n + 1);
	for (int i = 1 ; i <= n ; i++) cin>>pmt[i];
	vector<int> dfn(n + 1),fa(n + 1),in(n + 1),out(n + 1),inv(n + 1);
	int tot = 0;
	auto dfs = [&](auto &&self,int cur) -> void
	{
		tot++;
		dfn[tot] = cur;
		inv[cur] = tot;
		in[cur] = tot;
		for (const auto& i : g[cur])
		{
			if (fa[cur] != i)
			{
				fa[i] = cur;
				self(self,i);
			}
		}
		out[cur] = tot;
	};
	dfs(dfs,1);
	PersistentSegtree sgt(0,n);
	for (int i = 1 ; i <= n ; i++)
	{
		sgt.modify(in[pmt[i]],{1});
	}
	for (int i = 1 ; i <= q ; i++)
	{
		int l,r,x;
		cin>>l>>r>>x;
		int t = sgt.query_reg(l - 1,r,in[x],out[x]);
		if (t > 0)
		{
			cout<<"Yes\n";
		}
		else cout<<"No\n";
	}
	cout<<'\n';
	return;
}
```
