[题目喵](https://www.luogu.com.cn/problem/P1438)

像是这种第一次见的特殊处理我还是留一个好了喵，方便以后查看喵。

1. 区间拆分：当你要求在整个目标区间 $[L, R]$ 加上一个首项为 $A$，公差为 $D$ 的等差数列时，线段树会把这个区间拆分成好几个节点维护的子区间 $[l, r]$ 喵。
2. 首项偏移：对于被完全包含的某个子区间 $[l, r]$，它被加上的等差数列的公差仍然是 $D$，但它的首项已经不是 $A$ 了！因为前面已经走过了 $l - L$ 个元素，所以这个子区间的首项应该是 $A + (l - L) \times D$ 喵！这可是最容易出错的地方，笨蛋记住了没喵？！区间求和更新：给一个长度为 $len$ 的区间打上首项为 $first$，公差为 $D$ 的标记时，这个区间增加的总和根据等差数列求和公式就是：$len \times first + \frac{len \times (len - 1)}{2} \times D$ 喵。
3. 标记下放（Push Down）：这是重中之重喵！把当前节点 $[l, r]$ 的标记传给左右儿子时：
  - 左儿子：完美继承爸爸的首项和公差喵。
  - 右儿子：公差不变，但右儿子的起点比左儿子晚了，所以右儿子的首项要加上左儿子的长度乘以公差喵！

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
typedef __int128 i128;
typedef long double ld;
#define int long long
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;
const int MOD = 998244353;
const ll INF = 0x3f3f3f3f3f3f3f3f;
const int N = 1e6 + 10;

struct LazySegTree {
	struct Info {
		ll val = 0;
		Info operator+(const Info& x) const { return {val+x.val}; }
		void operator+=(const Info& x) { *this = *this + x; }
	};
	struct Lazy {
		ll a1 = 0,d = 0;
		Lazy operator+(const Lazy& x) { return {a1 + x.a1,d + x.d}; }
		void operator+=(const Lazy& x) { *this = *this + x; }
	};
	int min_, max_;
	vector<Info> info;
	vector<Lazy> tag;
	LazySegTree(int n): min_(1), max_(n), info(4 * n), tag(4 * n) {}
	LazySegTree(int l, int r): min_(l), max_(r), info(4 * (r - l + 1)), tag(4 * (r - l + 1)) {}
private:
	void operation(const Lazy& lazy, int i, int l, int r) {
		info[i].val += (r - l + 1) * lazy.a1 + (1 + r - l) * (r - l) / 2 *lazy.d;
		tag[i] += lazy;
	}
	void pull(int i) { info[i] = info[2 * i] + info[2 * i + 1]; }
	void push(int i, int l, int r) {
		if (tag[i].a1 == 0 && tag[i].d == 0) return;
		operation(tag[i], 2 * i, l, (l + r) / 2);
		auto t = tag[i];
		t.a1 = tag[i].a1 + ((l + r) / 2 - l + 1) * tag[i].d;
		operation(t, 2 * i + 1, (l + r) / 2 + 1, r);
		tag[i] = {};
	}
	void set(int pos, const Info& v, int i, int l, int r) {
		if (l == r) { info[i] = v; return; }
		push(i, l, r);
		if (pos <= (l + r) / 2) set(pos, v, 2 * i, l, (l + r) / 2);
		else set(pos, v, 2 * i + 1, (l + r) / 2 + 1, r);
		pull(i);
	}
	void rangeOperation(int x, int y, const Lazy& v, int i, int l, int r) {
		if (y < x || l > y || r < x) return;
		if (l >= x && r <= y) 
		{
			auto kk = v;
			kk.a1 = (l - x) * v.d + v.a1;
			return operation(kk, i, l, r);
		}
		push(i, l, r);
		rangeOperation(x, y, v, 2 * i, l, (l + r) / 2);
		rangeOperation(x, y, v, 2 * i + 1, (l + r) / 2 + 1, r);
		pull(i);
	}
	Info query(int x, int y, int i, int l, int r) {
		if (l >= x && r <= y) return info[i];
		push(i, l, r);
		if (y <= (l + r) / 2) return query(x, y, 2 * i, l, (l + r) / 2);
		if (x > (l + r) / 2) return query(x, y, 2 * i + 1, (l + r) / 2 + 1, r);
		return query(x, y, 2 * i, l, (l + r) / 2) + query(x, y, 2 * i + 1, (l + r) / 2 + 1, r);
	}
public:
	void set(int i, const Info& v) { set(i, v, 1, min_, max_); }
	void rangeOperation(int l, int r, const Lazy& v) { rangeOperation(l, r, v, 1, min_, max_); }
	Info query(int l, int r) { return query(l, r, 1, min_, max_); }
	friend ostream& operator<<(ostream& os, LazySegTree& seg) {
		os << "seg[";
		for (int i = seg.min_; i <= seg.max_; i++) {
			os << seg.query(i, i).val << ",";
		}
		os << "]";
		return os;
	}
};

void solve()
{
	int n,m;
	cin>>n>>m;
	LazySegTree sgt(n);
	for (int i = 1 ; i <= n ; i++)
	{
		int x;
		cin>>x;
		sgt.set(i,{x});
	}
	while(m--)
	{
		int k;
		cin>>k;
		if (k == 1)
		{
			int l,r,a1,d;
			cin>>l>>r>>a1>>d;
			sgt.rangeOperation(l,r,{a1,d});
		}
		else
		{
			int p;
			cin>>p;
			cout<<sgt.query(p,p).val<<'\n';
		}
	}
    return;
}
signed main()
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(0);
    int _ = 1;
    //std::cin>>_;
    while (_--)
    {
        solve();
    }
    return 0;
}
```
