# 主席树喵

对于这种问题，我们可以先找到一个传感器激活的条件是什么喵，激活的条件就是 **其所在的区间里面只有一个球是红的喵**，所以我们要满足的条件就是 $pre_x < l \leq x$ 并且 $x \leq r < next_x$ 喵，所以，我们就能发现，如果 **把l,r看成是二维平面上一个点的话，那么这个传感器被激活的条件就是它在矩阵 $(prex,x] \times [x,nextx)$ 这个矩形里面喵**

这里的 $(prex,x] \times [x,nextx)$ 其实就是横坐标范围为 (prex,x] 纵坐标范围为 [x,nextx) 的矩形喵

那么我们要用什么来维护二维平面上矩形的变化喵？

当一颗球变蓝之后，我们就要删掉其所在的那个球所对应的矩形里面传感器的权值喵，但是光删掉也不够喵，我们还要把他相邻两个球的区间扩大喵。然后把放进去的那个，加到答案里面喵。

那我们要用什么数据结构喵，那当然是用 **主席树** 喵。我们可以用主席树的版本号来当x轴的坐标喵，把主席树里面的当作是y轴的坐标喵，这样的话，我们就能很好地维护二维平面上的信息了喵

~~基于这个想法，我们可以用主席树维护二维前缀和喵，甚至不能带修喵~~

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
typedef __int128 i128;
typedef pair<int,int> pii;
typedef pair<ll,ll> pll;
const int MOD = 998244353;
const ll INF = 0x3f3f3f3f3f3f3f3f;
const int N = 1e6 + 10;
#define int ll

mt19937_64 rng(chrono::steady_clock::now().time_since_epoch().count());
ll rand(ll l, ll r) {return uniform_int_distribution<ll>(l, r)(rng);}

struct PersistentSegtree
{
    struct Info
    {
        ll val;
        Info operator+(const Info&x)
        {
            return {val + x.val};
        }
        void operator+=(const Info&x){*this = *this + x;}
    };
    const ll ie = 0;//所谓单位元
    //根节点应该是单位元喵如果是求和就是0，求max就是-inf
    struct Node
    {
        int ls,rs;
        Info v;
    };
    int max_,min_;
    vector<int> root;
    vector<Node> info;
    struct lazy {/* data */};
    PersistentSegtree(int l,int r):max_(r),min_(l),root(1){info.push_back({0,0,{ie}});}//数据范围喵
private:
    void pull(int p){info[p].v = info[info[p].ls].v + info[info[p].rs].v;}
    int set(int p,int pos,int l,int r,Info x)
    {
        if (!p)
        {
            p = info.size();
            info.emplace_back(0,0,Info{ie});           
        }
        if (l == r) {info[p].v = x;return p;}
        int mid = l + (r - l) / 2;
        if (pos <= mid) 
        {
            int newls = set(info[p].ls,pos,l,mid,x);
            info[p].ls = newls;
        }
        else 
        {
            int newrs = set(info[p].rs,pos,mid + 1,r,x);
            info[p].rs = newrs;
        }
        pull(p);
        return p;
    }
    int operation(int p,int pos,int l,int r,Info x)
    {
        auto t = info[p];
        info.emplace_back(t);
        p = info.size() - 1;
        if (l == r) {info[p].v += x;return p;}
        int mid = l + (r - l) / 2;
        if (pos <= mid)
        {
            int newls = operation(info[p].ls,pos,l,mid,x);
            info[p].ls = newls;
        }
        else
        {
            int newrs = operation(info[p].rs,pos,mid + 1,r,x);
            info[p].rs = newrs;
        }
        pull(p);
        return p;
    }
    Info query(int p,int l,int r,int ql,int qr)
    {
        if (!p || ql > qr) return {ie};
        int mid = l + (r - l) / 2;
        if (ql <= l && qr >= r) return info[p].v;
        if (qr <= mid) return query(info[p].ls,l,mid,ql,qr);
        else if (ql > mid) return query(info[p].rs,mid + 1,r,ql,qr);
        else return query(info[p].ls,l,mid,ql,qr) + query(info[p].rs,mid + 1,r,ql,qr);
    }
public:
    void set(int pos,Info val) {root[0] = set(root[0],pos,min_,max_,val);}
    int operation(int ver,int pos,Info val){return operation(ver,pos,min_,max_,val);}
	void modify(int pos,Info val) {
        int newroot = operation(root.back(),pos,min_,max_,val);
        root.emplace_back(newroot);
    }
    Info query(int q,int l,int r){return query(root[q],min_,max_,l,r);}
	int query_reg(int top,int bottom,int l,int r){return query(bottom,l,r).val - query(top,l,r).val;}
};

void solve()
{	
	ll n,m;
	cin>>n>>m;
	vector<int> l(n),r(n);
	ll sumn = 0;
	vector<vector<pll>> a(n + 1);
	PersistentSegtree sgt(-1,n + 1);
	for (int i = 1 ; i <= m ; i++) 
	{
		ll ls,rs;
		cin>>ls>>rs;
		a[ls].emplace_back(rs,i);
		if (ls == rs)
		{
			sumn += i * i;
		}
	}
	for (int i = 0 ; i < n ; i++)
	{
		int cur = sgt.root.back();	
		for (const auto& [rs,pos] : a[i])
		{
			cur = sgt.operation(cur,rs,{pos * pos});
		}
		sgt.root.emplace_back(cur);
	}
	for (int i = 0; i < n; i++)r[i] = i + 1;
	for (int i = n - 1; i >= 0; i--)l[i] = i - 1;
	cout<<sumn<<' ';
	for (int i = 0; i < n; i++)
	{
		ll opt;
		cin>>opt;
		int x = (opt + sumn) % n;
		int left1 = l[x];  // 真正的左邻居
		int right1 = r[x]; // 真正的右邻居
		int left2 = (left1 != -1) ? l[left1] : -1;
		int right2 = (right1 != n) ? r[right1] : n;

		// 注意版本查询：query_reg 需要传入正确的 top 和 bottom
		// 如果你想要版本范围 (top, bottom]，在这里对应 (left1, x]
		// 也就是 x 的版本 减去 left1 的版本
		sumn -= sgt.query_reg(left1 + 1, x + 1, x, right1 - 1); // x 失去的领地

		if (left1 != -1) { // 左边扩张
		sumn += sgt.query_reg(left2 + 1, left1 + 1, x, right1 - 1);
		}
		if (right1 != n) { // 右边扩张
		sumn += sgt.query_reg(left1 + 1, x + 1, right1, right2 - 1);
		}
		if (left1 != -1) r[left1] = right1;
		if (right1 != n) l[right1] = left1;

		cout<<sumn<<' ';
	}
	cout<<'\n';
	return;
}
```
