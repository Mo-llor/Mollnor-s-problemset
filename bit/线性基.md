struct LinearBasis
{
	int basis[64];
	vector<long long> p; // 用来存重构后的纯洁基底喵
    bool zero = false;   // 记录原集合能不能凑出 0 喵
	LinearBasis(){for (int i = 0 ; i < 64 ; i++)basis[i] = 0;}
	bool insert(int x)
	{
		for (int i = 63 ; i >= 0 ; i--)
		{
			if ((x >> i) & 1)
			{
				if (!basis[i])
				{
					basis[i] = x;
					return true;
				}
				x ^= basis[i];
			}
		}
		return false;
	}

	int MaxXor(int x)
	{
		int ans = x;
		for (int i = 63 ; i >= 0 ; i--)
		{
			if ((ans ^ basis[i]) > ans)
			{
				ans ^= basis[i];
			}
		}
		return ans;
	}

	// 重构线性基，把它变成对角矩阵喵！
    void rebuild(int total_n, int inserted_count) {
        // 如果成功插入线性基的元素个数小于总数，说明有废柴被消成了 0
        // 这就意味着原集合自己就能异或出 0 喵！
        zero = (inserted_count < total_n);
        
        for (int i = 63; i >= 0; --i) {
            for (int j = i - 1; j >= 0; --j) {
                // 如果 basis[i] 的第 j 位也是 1，就用 basis[j] 把它消掉喵！
                if ((basis[i] >> j) & 1) {
                    basis[i] ^= basis[j]; 
                }
            }
        }
        
        // 把非空的基底拿出来，从小到大排好队喵
        p.clear();
        for (int i = 0; i <= 63; ++i) {
            if (basis[i]) p.push_back(basis[i]);
        }
    }

    // 查询第 k 小的异或和喵
    long long queryKth(long long k) {
        if (zero) k--; // 如果能凑出 0，那么 0 就是名副其实的第 1 小，把 k 减 1 抵消掉喵
        if (k == 0) return 0;
        
        // 如果 k 大于能组合出的总数，说明根本没那么多组合，返回 -1 喵
        if (k >= (1LL << p.size())) return -1; 
        
        long long res = 0;
        for (int i = 0; i < p.size(); ++i) {
            // 把 k 看作二进制，第 i 位是 1 就选第 i 小的基底喵
            if ((k >> i) & 1) {
                res ^= p[i];
            }
        }
        return res;
    }
};
